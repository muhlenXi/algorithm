## Description

Given a non-negative integerÂ numRows, generate the first numRows of Pascal's triangle.

- [More info](https://leetcode-cn.com/problems/pascals-triangle/)

**Example:**

```
Input: 5
Output:
[
     [1],
    [1,1],
   [1,2,1],
  [1,3,3,1],
 [1,4,6,4,1]
]
```

## Solution


`Swift`

```swift

/// 118
func generate(_ numRows: Int) -> [[Int]] {
    var output = [[Int]]()
    for index in 0..<numRows {
        output.append(getRow(index))
    }
    
    return output
}

/// 119
func getRow(_ rowIndex: Int) -> [Int] {
    var output = [1]
    guard rowIndex > 0 else {
        return output
    }
    
    for _ in 0..<rowIndex {
        for i in 0..<output.count-1 {
            output[i] = output[i]+output[i+1]
        }
        output.insert(1, at: 0)
    }
    
    return output
}
```
