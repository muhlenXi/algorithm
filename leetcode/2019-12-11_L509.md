# 509. Fibonacci Number

`Swift`

```swift
/// 509
func fib(_ N: Int) -> Int {
    var caches = [Int: Int]()
    return fib(N, &caches)
}

func fib(_ n: Int, _ cache: inout[Int: Int]) -> Int{
    if let value = cache[n]{
        return value
    }
    
    var fibN = 0
    switch n {
    case 0:
        fibN = 0
    case 1, 2:
        fibN = 1
    default:
        fibN = fib(n-1, &cache) + fib(n-2, &cache)
    }
    
    cache[n] = fibN
    return fibN
}
```
