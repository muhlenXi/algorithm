# 67. Add Binary

### Date

- Edit by muhlenXi on 2019-12-22.

### Description

Given two binary strings, return their sum (also a binary string).

The input strings are both non-empty and contains only characters 1 orÂ 0.

**Example:**

```
Input: a = "11", b = "1"
Output: "100"

Input: a = "1010", b = "1011"
Output: "10101"
```


### Solution

`Swift`

```swift
func addBinary(_ a: String, _ b: String) -> String {
    var A = a.map { return $0 }
    var B = b.map { return $0 }
    let zeros = Array(repeating: Character("0"), count: abs(A.count-B.count))
    
    if A.count >= B.count {
        B.insert(contentsOf: zeros, at: 0)
    } else {
        A.insert(contentsOf: zeros, at: 0)
    }
    
    let one = Character("1")
    let zero = Character("0")
    
    var index = A.count - 1
    var carryBit = zero
    while index >= 0 {
        if A[index] == B[index]{
            let isEqualOne = A[index] == one
            A[index] = carryBit
            if isEqualOne {
                carryBit = one
            } else {
                carryBit = zero
            }
        } else {
            if carryBit == one {
                A[index] = zero
                carryBit = one
            } else {
                A[index] = one
                carryBit = zero
            }
        }
        index -= 1
    }
    
    if carryBit == one {
        A.insert(carryBit, at: 0)
    }
    
    return String(A)
}
```
