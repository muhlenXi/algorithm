func findRepeatNumber(_ nums: [Int]) -> Int {
    // ç»Ÿè®¡
    var counts = Array(repeating: 0, count: nums.count)
    for num in nums {
        counts[num] += 1
    }
    // éå†
    for (index, repeats) in counts.enumerated() {
        if repeats > 1 {
            return index
        }
    }
    
    return 0
}
