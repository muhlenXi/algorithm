func generateMatrix(_ n: Int) -> [[Int]] {
    var matrix = [[Int]]()
    guard n > 0 else {
        return matrix
    }
   
    for _ in 0..<n {
        matrix.append(Array(repeating: 0, count: n))
    }
    
    var left = 0
    var right = matrix[0].count-1
    var up = 0
    var down = matrix.count-1
    
    var row = 0
    var column = 0
    let total = matrix.count*matrix[0].count
    var index = 1
    while index <= total {
        // 右
        while column <= right {
            matrix[row][column] = index
            index += 1
            column += 1
        }
        column -= 1
        up += 1
        // 下
        row += 1
        while row <= down {
            matrix[row][column] = index
            index += 1
            row += 1
        }
        row -= 1
        column -= 1
        right -= 1
        // 左
        while column >= left {
            matrix[row][column] = index
            index += 1
            column -= 1
        }
        column += 1
        down -= 1
        // 上
        row -= 1
        while row >= up {
            matrix[row][column] = index
            index += 1
            row -= 1
        }
        
        row += 1
        column += 1
        left += 1
    }
    
    return matrix
}
