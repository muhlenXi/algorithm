class MyStack {
    
    private var inQueue = MyQueue()
    private var outQueue = MyQueue()

    /** Initialize your data structure here. */
    init() {
        
    }
    
    /** Push element x onto stack. */
    func push(_ x: Int) {
        inQueue.enqueue(x)
        
        let new = MyQueue()
        new.enqueue(x)
        while !self.outQueue.empty() {
            new.enqueue(self.outQueue.dequeue())
        }
        self.outQueue = new
    }
    
    /** Removes the element on top of the stack and returns that element. */
    func pop() -> Int {
        let num = outQueue.dequeue()
        
        let new = MyQueue()
        var count = 0
        while count < outQueue.count() {
            new.enqueue(self.inQueue.dequeue())
            count += 1
        }
        self.inQueue = new
        
        return num
    }
    
    /** Get the top element. */
    func top() -> Int {
        return outQueue.first() ?? 0
    }
    
    /** Returns whether the stack is empty. */
    func empty() -> Bool {
        return inQueue.empty()
    }
}

class MyQueue {
    
    private var list = [Int]()
    
    init() {
        
    }
    
    func enqueue(_ x: Int) {
        list.append(x)
    }
    
    func dequeue() -> Int {
        return list.removeFirst()
    }
    
    func empty() -> Bool {
        return list.isEmpty
    }
    
    func first() -> Int? {
        return list.first
    }
    
    func count() -> Int {
        return list.count
    }
}
