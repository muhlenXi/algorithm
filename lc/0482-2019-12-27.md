func licenseKeyFormatting(_ S: String, _ K: Int) -> String {
    guard K != 0 else {
        return ""
    }
    
    let characters = S
        .uppercased()
        .map { return $0 }
        .filter { $0 != Character("-") }
    var outputs = [String]()
    
    var index = characters.count-1
    while index >= 0 {
        let low = max(0, index-K+1)
        outputs.insert(String(characters[low...index]), at: 0)
        index -= K
    }
    
    return outputs.joined(separator: "-")
}
