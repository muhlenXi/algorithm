func queensAttacktheKing(_ queens: [[Int]], _ king: [Int]) -> [[Int]] {
    var board = [[Int]]()
    let n = 8
    for _ in 0..<n {
        board.append(Array(repeating: 0, count: 8))
    }
    
    for qu in queens {
        board[qu[0]][qu[1]] = 1
    }
    board[king[0]][king[1]] = 9
    
    var result = [[Int]]()
    for point in queens {
        if canAttack(&board, point) {
            result.append(point)
        }
    }
    return result
}

func canAttack(_ board: inout [[Int]], _ queens: [Int]) -> Bool {
    var isFind = false
    
    var row = queens[0]
    var col = queens[1]-1
    // 左
    while col >= 0 {
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        col -= 1
    }
    // 右
    row = queens[0]
    col = queens[1]+1
    while col < board.count {
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        col += 1
    }
    // 上
    row = queens[0]-1
    col = queens[1]
    while row >= 0 {
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        row -= 1
    }
    // 下
    row = queens[0]+1
    col = queens[1]
    while row < board.count {
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        row += 1
    }
    // 左上
    row = queens[0]-1
    col = queens[1]-1
    while row >= 0 && col >= 0{
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        row -= 1
        col -= 1
    }
    // 左下
    row = queens[0]+1
    col = queens[1]-1
    while row < board.count && col >= 0{
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        row += 1
        col -= 1
    }
    // 右上
    row = queens[0]-1
    col = queens[1]+1
    while row >= 0 && col < board.count {
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        row -= 1
        col += 1
    }
    // 右下
    row = queens[0]+1
    col = queens[1]+1
    while row < board.count && col < board.count {
        if board[row][col] == 9 {
            isFind = true
            break
        }
        if board[row][col] == 1 {
            break
        }
        row += 1
        col += 1
    }
    
    return isFind
}
