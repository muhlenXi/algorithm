func findTheDifference(_ s: String, _ t: String) -> Character {
    let ss = s.utf8CString
    let tt = t.utf8CString
    var sum: Int = 0
    
    for index in 0..<tt.count {
        sum += Int(tt[index])
        if index < ss.count {
            sum -= Int(ss[index])
        }
    }
    
    return Character(UnicodeScalar(sum)!)
}
