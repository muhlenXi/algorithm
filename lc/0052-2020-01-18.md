func totalNQueens(_ n: Int) -> Int {
    guard n > 0 else {
        return 0
    }
    
    var board = [[Int]]()
    for _ in 0..<n {
        board.append(Array(repeating: 0, count: n))
    }
    var result = 0
    backtrack(&board, 0, &result)
    return result
}

func backtrack(_ board: inout [[Int]], _ row: Int, _ result: inout Int) {
    if row == board.count {
        result += 1
        return
    }
    
    let columns = board[row].count
    for col in 0..<columns {
        if canPlaceQueen(board, row, col) == false {
            continue
        }
        
        board[row][col] = 1
        backtrack(&board, row+1, &result)
        board[row][col] = 0
    }
}

func canPlaceQueen(_ board: [[Int]], _ row: Int, _ cloumn: Int) -> Bool {
    let length = board.count
    // 判断同一列是否有 queen
    for row in 0..<length {
        if board[row][cloumn] == 1 {
            return false
        }
    }
    // 判断右斜对角线上是否有 queen
    var ro = row - 1
    var cl = cloumn + 1
    while ro >= 0 && cl < length {
        if board[ro][cl] == 1 {
            return false
        }
        ro -= 1
        cl += 1
    }
    // 判断左斜对角线上是否有 queen
    ro = row - 1
    cl = cloumn - 1
    while ro >= 0 && cl >= 0 {
        if board[ro][cl] == 1 {
            return false
        }
        ro -= 1
        cl -= 1
    }
    
    return true
}
