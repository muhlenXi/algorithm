func allCellsDistOrder(_ R: Int, _ C: Int, _ r0: Int, _ c0: Int) -> [[Int]] {
    var map = [Int: [[Int]]]()
    for r in 0..<R {
        for c in 0..<C {
            let dis = abs(r-r0) + abs(c-c0)
            if var val = map[dis] {
                val.append([r, c])
                map[dis] = val
            } else {
                map[dis] = [[r, c]]
            }
        }
    }
    
    var result = [[Int]]()
    let keys = map.keys.sorted()
    for key in keys {
        result.append(contentsOf: map[key]!)
    }
    
    return result
}
