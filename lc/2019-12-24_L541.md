func reverseStr(_ s: String, _ k: Int) -> String {
    var maxCount = s.count / k
    if s.count % k != 0 {
        maxCount += 1
    }
    
    var ss = s.map { return $0 }
    var index = 0
    var flag = 0
    
    while index < ss.count {
        if flag % 2 == 0 {
            var low = index
            var high = min(index+k-1, ss.count-1)
            while low <= high {
                let temp = ss[low]
                ss[low] = ss[high]
                ss[high] = temp
                low += 1
                high -= 1
            }
        }
        index += k
        flag += 1
    }
    
    return String(ss)
}
