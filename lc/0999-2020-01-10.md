func numRookCaptures(_ board: [[Character]]) -> Int {
    let R = Character("R")
    let p = Character("p")
    let B = Character("B")
    var rx = 0
    var ry = 0
    for y in 0..<board.count {
        if board[y].contains(R) {
            for x in 0..<board[y].count {
                if board[y][x] == R {
                    rx = x
                    ry = y
                    break
                }
            }
        }
    }
    
    var count = 0
    // up
    var up = ry
    var upCatch = false
    while up >= 0 && board[up][rx] != B && upCatch == false{
        if board[up][rx] == p {
            count += 1
            upCatch = true
        }
        up -= 1
    }
    // down
    var down = ry
    var downCatch = false
    while down < 8 && board[down][rx] != B  && downCatch == false{
        if board[down][rx] == p {
            count += 1
            downCatch = true
        }
        down += 1
    }
    // left
    var left = rx
    var leftCatch = false
    while left >= 0 && board[ry][left] != B && leftCatch == false {
        if board[ry][left] == p {
            count += 1
            leftCatch = true
        }
        left -= 1
    }
    // right
    var right = rx
    var rightCatch = false
    while right < 8 && board[ry][right] != B && rightCatch == false {
        if board[ry][right] == p {
            count += 1
            rightCatch = true
        }
        right += 1
    }
    
    return count
}
