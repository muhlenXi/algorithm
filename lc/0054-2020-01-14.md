func spiralOrder(_ matrix: [[Int]]) -> [Int] {
    guard matrix.count > 0 else {
        return [Int]()
    }
    
    var result = [Int]()
    
    var left = 0
    var right = matrix[0].count-1
    var up = 0
    var down = matrix.count-1
    
    var row = 0
    var column = 0
    let total = matrix.count*matrix[0].count
    while result.count < total {
        // 右
        while column <= right {
            result.append(matrix[row][column])
            column += 1
        }
        column -= 1
        up += 1
        // 下
        row += 1
        while row <= down {
            result.append(matrix[row][column])
            row += 1
        }
        row -= 1
        column -= 1
        right -= 1
        // 左
        while column >= left {
            if !result.contains(matrix[row][column]) {
                result.append(matrix[row][column])
            }
            column -= 1
        }
        column += 1
        down -= 1
        // 上
        row -= 1
        while row >= up {
            if !result.contains(matrix[row][column]) {
                result.append(matrix[row][column])
            }
            row -= 1
        }
        
        row += 1
        column += 1
        left += 1
    }
    
    return result
}
