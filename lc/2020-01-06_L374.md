public int guessNumber(int n) {
    int low = 1;
    int high = n;
    while(low <= high) {
        int mid = low + (high-low)/2;
        int guessMid = guess(mid);
        if(guessMid == 0) {
            return mid;
        } else if(guessMid == -1) {
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }
    return 0;
}

public int guess(int num) {
    int target = 1;
    if(target > num) {
        return 1;
    } else if(num == target) {
        return 0;
    } else {
        return -1;
    }
}
