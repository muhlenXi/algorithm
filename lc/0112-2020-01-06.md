# 112. 路径总和

### 简述

- [Leetcode](https://leetcode-cn.com/problems/path-sum/)

### 思路

- 深度优先遍历，是叶子节点则计算路径和

### 代码

`Swift`

func hasPathSum(_ root: TreeNode?, _ sum: Int) -> Bool {
    guard root != nil else {
        return false
    }
    
    if sum - root!.val == 0 {
        if root!.left == nil && root!.right == nil {
            return true
        } else {
            return false
        }
    }
    
    var pathNumbers = 0
    if sum - root!.val == 0 {
        pathNumbers += 1
    }
    sumOfPath(root!.left, sum - root!.val, &pathNumbers)
    sumOfPath(root!.right, sum - root!.val, &pathNumbers)
    
    return pathNumbers > 0
}

// 前序遍历 统计满足 sum 的路径个数
func sumOfPath(_ root: TreeNode?, _ sum: Int, _ pathNumbers: inout Int) {
    guard root != nil else {
        return
    }
    
    if sum - root!.val == 0 && root!.left == nil && root!.right == nil {
        pathNumbers += 1
    }
    sumOfPath(root!.left, sum - root!.val, &pathNumbers)
    sumOfPath(root!.right, sum - root!.val, &pathNumbers)
}
