func change(_ amount: Int, _ coins: [Int]) -> Int {
    if amount == 0 && coins.count == 0 {
        return 1
    }
    if amount == 0 {
        return 1
    }
    if coins.count == 0 {
        return 0
    }
    
    var dp = Array(repeating: 0, count: amount+1)
    dp[0] = 1
    
    for ele in coins {
        for a in ele...amount {
            dp[a] += dp[a-ele]
        }
    }
    
    return dp[amount]
}

func change1(_ amount: Int, _ coins: [Int]) -> Int {
    var temp = [Int]()
    var tempSum = 0
    var result = 0
    dfs(amount, coins, &temp, &tempSum, 0, &result)
    return result
}

func dfs(_ amount: Int,
        _ coins: [Int],
        _ temp: inout [Int],
        _ tempSum: inout Int,
        _ index: Int,
        _ result: inout Int) {
    if tempSum == amount {
        result += 1
        print(temp)
        return
    }
    if tempSum > amount {
        return
    }
    
    for i in index..<coins.count {
        let coin = coins[i]
        temp.append(coin)
        tempSum += coin
        
        dfs(amount, coins, &temp, &tempSum, i, &result)
        
        tempSum -= coin
        temp.removeLast()
    }
}
