func partition(_ head: ListNode?, _ x: Int) -> ListNode? {
    guard let h = head else {
        return head
    }
    
    let target = x
    
    let aHead: ListNode? = ListNode(110)
    let bHead: ListNode? = ListNode(120)
    
    var current: ListNode? = h
    var ca: ListNode?  = aHead
    var cb: ListNode?  = bHead
    
    while current != nil {
        let v = current!.val
        let node = ListNode(v)
        if v < target {
            ca?.next = node
            ca = ca?.next
        } else {
             cb?.next = node
             cb = cb?.next
        }
        current = current?.next
    }
    
    ca?.next = bHead?.next
    
    return aHead?.next
}

func partition1(_ head: ListNode?, _ x: Int) -> ListNode? {
    guard let h = head else {
        return head
    }
    
    let target = x
    var a = [Int]()
    var b = [Int]()
    
    var current: ListNode? = h
    while current != nil {
        let v = current!.val
        if v < target {
            a.append(v)
        } else {
            b.append(v)
        }
        current = current?.next
    }
    
    let numbers = a + b
    current = h
    for element in numbers {
        current?.val = element
        current = current?.next
    }
    
    return head
}
