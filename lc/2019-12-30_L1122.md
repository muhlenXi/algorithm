func relativeSortArray(_ arr1: [Int], _ arr2: [Int]) -> [Int] {
    var list = arr1
    var table = [Int: Int]()
    for element in arr1 {
        if let val = table[element] {
            table[element] = val + 1
        } else {
            table[element] = 1
        }
    }
    
    var index = 0
    for element in arr2 {
        let val = table[element] ?? 0
        if val > 0 {
            table[element] = nil
            for _ in 0..<val {
                list[index] = element
                index += 1
            }
        }
    }
    
    var remain = [Int]()
    for element in table {
        for _ in 0..<element.value {
            remain.append(element.key)
        }
    }
    remain.sort()
    for element in remain {
        list[index] = element
        index += 1
    }
    
    return list
}
