# 63. 不同路径 II

### 简述

- [Leetcode](https://leetcode-cn.com/problems/unique-paths-ii/)

### 思路

- 动态规划

### 代码

`Swift`

```swift
// 63 DP
func uniquePathsWithObstacles(_ obstacleGrid: [[Int]]) -> Int {
    // 最后一个元素是障碍
    if let lastRow = obstacleGrid.last,
        let last = lastRow.last,
        last == 1 {
        return 0
    }
    
    let m = obstacleGrid.count
    let n = obstacleGrid[0].count
    var dp = [[Int]]()
    for _ in 0..<m {
        dp.append(Array(repeating: 0, count: n))
    }
    dp[0][0] = 1
    
    // 第0行
    var column = 0
    while column < n && obstacleGrid[0][column] == 0 {
        column += 1
    }
    for j in 0..<n {
        dp[0][j] = 1
        if j >= column {
            dp[0][j] = 0
        }
    }
    // 第0列
    var row = 0
    while row < m && obstacleGrid[row][0] == 0 {
        row += 1
    }
    for i in 0..<m {
        dp[i][0] = 1
        if i >= row {
            dp[i][0] = 0
        }
    }
    
    for i in 1..<m {
        for j in 1..<n {
            if obstacleGrid[i-1][j] != 1 {
               dp[i][j] += dp[i-1][j]
            }
            if obstacleGrid[i][j-1] != 1 {
               dp[i][j] += dp[i][j-1]
            }
        }
    }
    
    return dp[m-1][n-1]
}
```
