func checkPerfectNumber(_ num: Int) -> Bool {
    guard num > 0 else { return false }
    
    var sum = 1
    let high = Int(sqrt(Double(num)))
    guard high >= 2 else { return false }
    for index in 2...high{
        if num % index == 0 {
            let val = num/index
            if val == index {
                sum += val
            } else {
                sum += (index + val)
            }
        }
    }
    return sum == num
}
