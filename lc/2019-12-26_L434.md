func countSegments(_ s: String) -> Int {
    var cc = s.trimmingCharacters(in: .whitespaces).utf8CString
    cc.removeLast()
    guard cc.count != 0 else { return 0 }

    var count = 1
    var index = 0
    while index < cc.count {
        if cc[index] == 32 {
            count += 1
            while index+1 < cc.count && cc[index+1] == 32 {
                index += 1
            }
        }
        index += 1
    }
    
    return count
}
