# 216. 组合总和 III

### 简述

- [Leetcode](https://leetcode-cn.com/problems/combination-sum-iii/)

### 思路

- 回溯算法

### 代码

`Swift`

```swift
// 216
func combinationSum3(_ k: Int, _ n: Int) -> [[Int]] {
    var nums = [Int]()
    for ele in 1...9 {
        nums.append(ele)
    }
    
    var result = [[Int]]()
    var temp = [Int]()
    var tempSum = 0
    backtrack(nums, n, k, &temp, &tempSum, 0, &result)
    return result
}

// 从当前以后的元素中进行回溯
func backtrack(_ candidates: [Int],
        _ target: Int,
        _ k: Int,
        _ temp: inout [Int],
        _ tempSum: inout Int,
        _ index: Int,
        _ result: inout [[Int]]) {
    if index > candidates.count {
        return
    }
    if tempSum == target, temp.count == k {
        result.append(temp)
        return
    }
    if tempSum > target {
        return
    }
    
    for i in index..<candidates.count {
        // 去重
        if i > index && candidates[i-1] == candidates[i] {
            continue
        }
        
        let ele = candidates[i]
        temp.append(ele)
        tempSum += ele
        backtrack(candidates, target, k, &temp, &tempSum, i+1, &result)
        temp.removeLast()
        tempSum -= ele
    }
}
```
