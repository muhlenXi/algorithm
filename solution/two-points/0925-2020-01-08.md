func isLongPressedName(_ name: String, _ typed: String) -> Bool {
    let names = name.map {return $0}
    let types = typed.map {return $0}
    if names.count > types.count {
        return false
    }
    
    var n = 0
    var t = 0
    while n < names.count && t < types.count{
        let ne = names[n]
        var count0 = 0
        while n < names.count && names[n] == ne{
            n += 1
            count0 += 1
        }
        n -= 1
        
        var te = types[t]
        var count1 = 0
        while t < types.count && types[t] == te{
            t += 1
            count1 += 1
        }
        t -= 1
        te = types[t]
        if ne != te {
            return false
        }
        if count0 > count1 {
            return false
        }
        
        n += 1
        t += 1
    }
    
    if n != names.count {
        return false
    }
    if t != types.count {
        return false
    }
    
    return true
}
