// 938
func rangeSumBST(_ root: TreeNode?, _ L: Int, _ R: Int) -> Int {
    guard root != nil else {
        return 0
    }
    var sum = 0
    travel(root, L, R, &sum)
    return sum
}

//
func travel(_ root: TreeNode?, _ L: Int, _ R: Int, _ sum: inout Int) {
    guard root != nil else {
        return
    }
    
    travel(root?.left, L, R, &sum)
    if root!.val >= L && root!.val <= R {
        sum += root!.val
    }
    travel(root?.right, L, R, &sum)
}
