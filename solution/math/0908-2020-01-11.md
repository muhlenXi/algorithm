func smallestRangeI(_ A: [Int], _ K: Int) -> Int {
    let sorted = A.sorted()
    let max = sorted.last!
    let min = sorted.first!
    if (max-K) - (min+K) >= 0 {
        return max - K - min - K
    }
    
    return 0
}
