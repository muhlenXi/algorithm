func numPrimeArrangements(_ n: Int) -> Int {
    var table = Array(repeating: 0, count: n)
    for index in 0..<n {
        table[index] = index+1
    }
    
    table[0] = 0
    var index = 1
    while index < n {
        if table[index] != 0 {
            var i = index + table[index]
            while i < n {
                table[i] = 0
                i += table[index]
            }
        }
        index += 1
    }
    
    var primeCount = 0
    var normalCount = 0
    for element in table {
        if element == 0 {
            normalCount += 1
        } else {
            primeCount += 1
        }
    }
    
    var total = 1
    let low = min(primeCount, normalCount)
    for index in 1...max(primeCount, normalCount) {
        if index <= low {
            total *= (index * index)
        } else {
            total *= index
        }
        total = total % 1000000007
    }
    
    return total
}
