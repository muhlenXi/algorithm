func oddEvenList(_ head: ListNode?) -> ListNode? {
    guard let h = head else {
        return head
    }
    
    let aHead: ListNode? = h
    var ac: ListNode? = h
    let bHead: ListNode? = h.next
    var bc: ListNode? = h.next
    
    var index = 3
    var current: ListNode? = bHead?.next
    while current != nil {
        let v = current!.val
        let node = ListNode(v)
        if index % 2 == 1 {
            ac?.next = node
            ac = ac?.next
        } else {
            bc?.next = node
            bc = bc?.next
            
        }
        current = current?.next
        index += 1
    }
    
    bc?.next = nil
    ac?.next = bHead
    
    return aHead
}
